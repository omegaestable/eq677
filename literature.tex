\begin{problem}
Let $M$ be a finite nonempty set with a binary operation $\op$ satisfying
\[
x=y\op\bigl(x\op((y\op x)\op y)\bigr)
\qquad (x,y\in M). \tag{H}
\]
Prove
\[
x=((x\op x)\op x)\op x
\qquad (x\in M). \tag{G}
\]
\end{problem}

\section*{Hints}

\begin{enumerate}[label=\textbf{\arabic*.}]
\item For $a\in M$, set $L_a(t)=a\op t$.  From (H),
\[
b=a\op\bigl(b\op((a\op b)\op a)\bigr).
\]
Hence $L_a$ is bijective.  Write $a\backslash b$ for the unique $u$ with $a\op u=b$;
\[
a\backslash b=b\op((a\op b)\op a). \tag{D}
\]

\item For $y,z\in M$,
\[
z=(y\op z)\op\bigl((y\op(y\op z))\op y\bigr), \tag{T}
\]
and
\[
(y\op(y\op z))\op y
=z\op\bigl(((y\op z)\op z)\op(y\op z)\bigr). \tag{K}
\]
Use (H) with $x=y\op z$, then compare with (H) after $y\mapsto y\op z$.

\item Fix $x\in M$ and put
\[
a=x\op x,
\qquad
q=a\op x=((x\op x)\op x),
\qquad
p=x\backslash x.
\]
The goal at $x$ is
\[
q\op x=x. \tag{1}
\]
If $u\op x=x$, then $u=q$.  Also (1) is equivalent to each of
\[
q\backslash x=x,
\qquad
((x\op x)\op q)\op(x\op x)=x,
\qquad
(q\op x)\op q=p. \tag{2}
\]

\item Define
\[
H(t)=(x\op t)\op x,
\qquad
F(t)=x\op(t\op x).
\]
Prove
\[
x\backslash t=t\op H(t),
\qquad
F(L_x(t))=L_x(H(t)). \tag{3}
\]
Thus $F$ and $H$ are conjugate.  A fixed point of either map is equivalent to (1), and
is unique.

\item Let
\[
c_0=x,
\qquad
c_{k+1}=x\op c_k.
\]
Let $d$ be the least period of this $L_x$-orbit, read subscripts modulo $d$, and set
\[
d_k=c_k\op x.
\]
\[
c_{k-1}=c_k\op d_{k+1}. \tag{4}
\]
\[
p=c_{d-1},
\qquad
q=d_1=c_{d-2},
\qquad
p\op c_1=q,
\qquad
d_j=x\Rightarrow j\equiv d-2\pmod d. \tag{5}
\]

\item If $d=1$, then (1).  If $d=2$, contradiction.  If $d=3$, write
\[
c_0=x,
\qquad
c_1=a=q,
\qquad
c_2=p.
\]
\[
x=a\op(p\op x),
\qquad
a\op p=x,
\qquad
p\op x=p,
\]
then
\[
(a\op a)\op a=p
\qquad\text{and}\qquad
(a\op a)\op a=x\op(a\op a).
\]
Then $a\op a=a$, hence $p=a$, contradiction.  Thus any unresolved $x$ has $d\ge4$.

\item Assume $q\op x\ne x$.  Then $u\op x\ne x$ for all $u\in M$.  Since $M$ is finite,
there are $y\ne z$ and $e\in M$ with
\[
y\op x=z\op x=e. \tag{6}
\]
From (T),
\[
(y\op e)\op y=(z\op e)\op z=e\backslash x. \tag{7}
\]
Moreover $y\op e=z\op e\Rightarrow y=z$.

\item Prove the collision lemma
\[
\boxed{
y\op x=z\op x=e,\quad y\ne z
\quad\Longrightarrow\quad
e\op x=x.}
\tag{C}
\]
Then (1) follows.  Define
\[
\Phi(u,v)=\bigl(u\op v,
(u\op(u\op v))\op u\bigr).
\]
\[
\Phi(u,v)=(r,s)\Rightarrow
r\op s=v. \tag{8}
\]
\[
\Phi^n(u,v)=(\alpha_n,\alpha_{n+2}),
\qquad
\alpha_n\op\alpha_{n+2}=\alpha_{n+1}. \tag{9}
\]
\[
t=(y\op e)\op y=(z\op e)\op z,
\qquad
e\op t=x,
\qquad
\Phi(y,x)=\Phi(z,x)=(e,t). \tag{10}
\]
Use the first repeated pair in the finite forward $\Phi$-orbit of $(e,t)$ and pull it
back through (9).  It suffices to force $t=x$.

\item Optional: if $d=4$, write
\[
c_0=x,
\qquad
c_1=a,
\qquad
c_2=q,
\qquad
c_3=p,
\qquad
r=q\op x,
\qquad
s=p\op x.
\]
\[
x\op a=q,
\qquad
a\op x=q,
\qquad
a\op r=x,
\qquad
x\op q=p,
\qquad
x\op p=x,
\qquad
p\op a=q,
\]
and
\[
r\notin\{x,a,q\}. \tag{11}
\]
If $r=p$, show further that
\[
q\op a=q,
\qquad
s=(q\op q)\op q=a\op(q\op q),
\qquad
s\notin\{x,a,q,p\}. \tag{12}
\]
\end{enumerate}

\begin{goal}
For arbitrary $x$, prove (C), (1), or any identity in (2).  Then (G) holds for all
$x\in M$.
\end{goal}
