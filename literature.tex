\begin{abstract}
Checked local facts for finite magmas satisfying
\[
\E{677}:\quad x=y\op\bigl(x\op((y\op x)\op y)\bigr),
\qquad
\E{255}:\quad x=((x\op x)\op x)\op x.
\]
Fix \(x\).  Put \(a=x\op x\), \(q=a\op x\), and \(p=x\backslash x\).  The target is
\[
\EqAt{\E{255}}{x}\quad\Longleftrightarrow\quad q\op x=x.
\]
Everything below is proved here from \(\E{677}\) plus finiteness, or marked as external
ETP input.  No right cancellation, associativity, identity element, or quasigroup
structure is used.
\end{abstract}

\section{Current state roadmap}

\paragraph{Status legend.}
\emph{Local} means proved in this note from \(\E{677}\), finiteness, and left
cancellation supplied by Proposition~\ref{prop:left-division}.  \emph{External} means
input from the Equational Theories Project (ETP).  \emph{Computational} means a
proof-directed finite search or public database check, useful as calibration but not a
line-by-line proof.  \emph{Open} means a named gate whose closure would advance or finish
the finite implication.

\paragraph{One-line target.}
For a fixed point \(x\), put
\[
a=x\op x,\qquad q=a\op x,\qquad p=x\backslash x.
\]
The finite implication is reduced locally to any one of the equivalent targets
\[
\begin{aligned}
\EqAt{\E{255}}{x}
&\Longleftrightarrow q\op x=x
\Longleftrightarrow q\backslash x=x\\
&\Longleftrightarrow ((x\op x)\op q)\op(x\op x)=x
\Longleftrightarrow (q\op x)\op q=p.
\end{aligned}
\]

\paragraph{Trusted local package.}
The note proves finite left division, the transformed identity (T), the key identity
(K), the unique right fixed-point witness, fixed-map equivalences, the principal
\(L_x\)-orbit recurrence, the impossibility of periods \(2\) and \(3\), right-collision
rectangles, and the current period-four gate.  In a bad finite counterexample at \(x\),
the right column \(R_x\) misses \(x\), hence has a nontrivial collision fiber; on every
such fiber the splitter \(t\mapsto t\op e\) is injective and satisfies the rectangle law
\[
(t\op e)\op t=e\backslash x.
\]

\paragraph{External and computational context.}
ETP input says the unrestricted implication \(\E{677}\models\E{255}\) is false, so a
finite proof must visibly use finiteness.  ETP also records the linear model families
and the obstruction to affine-fiber counterexamples over positive linear bases.  The
public Equation 677 database, checked live on 2026-05-21, contains \(841\) records, all
marked as satisfying \(\E{255}\); \(116\) of these are non-right-cancellative positive
examples and therefore the useful near-miss pool.  Solver evidence has closed all
bad-witness searches through order \(9\), and at order \(10\) has closed periods \(4\),
\(5\), \(6\), \(8\), and \(9\); the broad frontier remains periods \(7\) and \(10\).
These searches guide the gates below but are not used as proof.

\paragraph{Open gates.}
The active route is to prove \(q\op x=x\).  The sharpest current gates are:
the expert collision pullback \(a\op x=b\op x=e,\ a\neq b\Rightarrow e\op x=x\);
the splitter-counting gate for a right collision fiber; the period-four external gate;
and the period-five pruning gate.  All are stated precisely at the end of the note.

\paragraph{Guardrails.}
No argument below assumes right cancellation, associativity, an identity element,
quasigroup or group structure, universal idempotence, the retired quotient family
\(d_k\backslash x=c_{k-2}\op c_{k-1}\), or the retired full shift
\(p\op d_k=d_{k+1}\).  Every cancellation is left cancellation by a proved bijective
left translation.

\section{Setup and finite tools}

\emph{Status: local finite proof package, with the ETP warning in
Remark~\ref{rem:external-guardrails} marked external.}

A magma is a set with one binary operation \(\op\).  For \(u\in M\), write
\[
L_u(t)=u\op t,\qquad R_u(t)=t\op u,\qquad S(t)=t\op t.
\]
If \(L_u\) is bijective, \(u\backslash v\) denotes the unique \(w\) with \(u\op w=v\).
All indices on an \(L_x\)-cycle are read modulo its period.

\begin{proposition}[Finite left division]\label{prop:left-division}
In every finite \(\E{677}\)-magma, each \(L_a\) is bijective and
\[
a\backslash b=b\op((a\op b)\op a).
\]
Also
\[
a\backslash(a\backslash b)=(a\backslash b)\op(b\op a),
\qquad
b\backslash(a\backslash b)=(a\op b)\op a.
\]
\end{proposition}
\begin{proof}
\cert \(\E{677}\) gives
\[
b=a\op\bigl(b\op((a\op b)\op a)\bigr),
\]
so \(L_a\) is surjective, hence bijective by finiteness.  Substitute \(a\backslash b\)
for \(b\), and use uniqueness of left division.
\end{proof}

\begin{lemma}[Two global identities]\label{lem:transformed-key}
For all \(y,z\),
\[
z=(y\op z)\op((y\op(y\op z))\op y) \tag{T}
\]
and
\[
(y\op(y\op z))\op y
=z\op(((y\op z)\op z)\op(y\op z)). \tag{K}
\]
\end{lemma}
\begin{proof}
\cert For (T), substitute \(y\op z\) for the law-variable \(x\) in \(\E{677}\), then
left-cancel \(y\).  For (K), compare (T) with \(\E{677}\) using parameter \(y\op z\),
then left-cancel \(y\op z\).
\end{proof}

\begin{remark}[External ETP guardrails]\label{rem:external-guardrails}
The unrestricted implication \(\E{677}\models\E{255}\) is false in the free
\(\E{677}\)-magma, so any proof must use finiteness.  Linear models satisfying
\(\E{677}\) satisfy \(\E{255}\); affine-fiber extensions over them are ruled out by ETP
structure theory.  Finite \(\E{677}\)-magmas need not be right-cancellative
\cite{ETP677}.
\end{remark}

\section{Local equivalences}

\emph{Status: locally proved reductions from the global implication to the fixed-point
gate \(q\op x=x\).}

Fix a finite \(\E{677}\)-magma \(M\) and \(x\in M\).  Put
\[
a=x\op x,\qquad q=a\op x=(x\op x)\op x,\qquad p=x\backslash x.
\]

\begin{proposition}[Unique right fixed-point witness]\label{prop:unique-witness}
If \(u\op x=x\), then \(u=q\).  Hence
\[
\EqAt{\E{255}}{x}
\quad\Longleftrightarrow\quad
\exists u\ (u\op x=x)
\quad\Longleftrightarrow\quad
q\op x=x.
\]
\end{proposition}
\begin{proof}
\cert If \(u\op x=x\), then (T) gives \(x=x\op(x\op u)\).  The specialization
\((x,y)=(x,x)\) in \(\E{677}\) gives \(x=x\op(x\op q)\).  Two left cancellations give
\(u=q\).
\end{proof}

\begin{proposition}[Equivalent local targets]\label{prop:local-targets}
The following are equivalent:
\[
\EqAt{\E{255}}{x},\qquad
q\op x=x,\qquad
q\backslash x=x,\qquad
((x\op x)\op q)\op(x\op x)=x,\qquad
(q\op x)\op q=p.
\]
The fourth identity is the identity sometimes called Lemma L.
\end{proposition}
\begin{proof}
\cert The first two are Proposition~\ref{prop:unique-witness}.  (T), with
\((y,z)=(x\op x,x)\), gives
\[
x=q\op(((x\op x)\op q)\op(x\op x)).
\]
Thus \(q\backslash x=((x\op x)\op q)\op(x\op x)\).  Finally
\[
q\backslash x=x\op((q\op x)\op q),
\]
and left cancellation by \(x\) compares this with \(p=x\backslash x\).
\end{proof}

\begin{lemma}[Fixed maps]\label{lem:fixed-map}
Define
\[
H_x(t)=((x\op t)\op x),\qquad F_x(t)=x\op(t\op x).
\]
Then
\[
x\backslash t=t\op H_x(t),\qquad F_x(L_x(t))=L_x(H_x(t)).
\]
Thus \(F_x\) and \(H_x\) have the same finite cycle structure.  Moreover
\(\EqAt{\E{255}}{x}\) is equivalent to either map having a fixed point.  If fixed points
exist, they are unique:
\[
\operatorname{Fix}(F_x)=\{x\backslash q\},\qquad
\operatorname{Fix}(H_x)=\{x\backslash(x\backslash q)\}.
\]
Under \(q\op x\neq x\), both maps are fixed-point-free.
\end{lemma}
\begin{proof}
\cert \(\E{677}\) with parameter \(x\) gives
\[
t=x\op(t\op((x\op t)\op x))=x\op(t\op H_x(t)),
\]
hence \(x\backslash t=t\op H_x(t)\).  Also
\[
F_x(L_x(t))=x\op((x\op t)\op x)=x\op H_x(t)=L_x(H_x(t)),
\]
so \(F_x=L_xH_xL_x^{-1}\), and \(F_x,H_x\) are conjugate.

Suppose \(F_x(t)=t\).  Comparing
\[
x\op(t\op x)=t=x\op(t\op H_x(t))
\]
and left-cancelling first \(x\) and then \(t\), we get \(H_x(t)=x\).  Thus
\((x\op t)\op x=x\), so Proposition~\ref{prop:unique-witness} gives \(x\op t=q\);
hence \(t=x\backslash q\).  Conversely, if \(q\op x=x\) and \(t=x\backslash q\), then
\(H_x(t)=(x\op t)\op x=q\op x=x\), and the displayed edge formula gives
\[
t=x\op(t\op x)=F_x(t).
\]
Thus \(F_x\) has a fixed point exactly when \(\EqAt{\E{255}}{x}\) holds, and then its
only fixed point is \(x\backslash q\).

The statement for \(H_x\) follows by conjugacy: \(H_x(s)=s\) iff
\(F_x(L_x(s))=L_x(s)\).  Hence, when fixed points exist,
\[
L_x(s)=x\backslash q,\qquad s=x\backslash(x\backslash q).
\]
If \(q\op x\neq x\), Proposition~\ref{prop:unique-witness} says that no right fixed
point over \(x\) exists, so neither \(F_x\) nor \(H_x\) has a fixed point.
\end{proof}

\begin{lemma}[Division return trap]\label{lem:division-trap}
For all \(y,z\), put
\[
u=y\op z,\qquad v=u\op y.
\]
Then
\[
v=z\backslash(y\backslash z),\qquad y\backslash z=z\op v. \tag{D}
\]
Consequently, if also \(z\op v=u\), then
\[
z=y\op u.
\]
\end{lemma}
\begin{proof}
\cert The formula for \(y\backslash z\) is just the left-division form of \(\E{677}\):
\[
y\backslash z=z\op((y\op z)\op y)=z\op v.
\]
Left-dividing this equality by \(z\) gives
\[
v=z\backslash(y\backslash z).
\]
If \(z\op v=u\), then the displayed identity gives \(y\backslash z=u\).  Left-multiply
by \(y\) to obtain \(z=y\op u\).
\end{proof}

\section{Orbit facts}

\emph{Status: locally proved orbit bookkeeping and the current period-four gate.}

Define
\[
c_0=x,\qquad c_{k+1}=x\op c_k,
\]
and let \(d\) be the minimal period of the \(L_x\)-orbit of \(x\).  Put
\[
d_k=c_k\op x.
\]

\begin{proposition}[Basic orbit package]\label{prop:basic-orbit}
For all \(k\),
\[
c_{k-1}=c_k\op d_{k+1},\qquad d_1=c_{d-2},
\]
\[
x\backslash c_k=c_{k-1},\qquad c_k\backslash c_{k-1}=d_{k+1}.
\]
In particular
\[
p=c_{d-1},\qquad q=x\backslash p=d_1=c_{d-2}=a\op x,
\qquad p\op a=q,\qquad p\backslash q=a.
\]
If \(d_j=x\), then \(j\equiv d-2\pmod d\).
\end{proposition}
\begin{proof}
\cert Applying \(\E{677}\) with parameter \(x\) and law-variable \(c_k\) gives
\[
c_k=x\op\bigl(c_k\op((x\op c_k)\op x)\bigr)
   =x\op(c_k\op d_{k+1}).
\]
Since also \(c_k=x\op c_{k-1}\), left cancellation by \(x\) gives
\[
c_{k-1}=c_k\op d_{k+1}.
\]
The equality \(x\op q=p\) follows from the specialization
\[
x=x\op(x\op q)
\]
of \(\E{677}\) at \((x,y)=(x,x)\), because \(p=x\backslash x\).  Therefore
\(q=x\backslash p=c_{d-2}\), while \(q=a\op x=d_1\) by definition.  The division
statements follow from \(x\op c_{k-1}=c_k\) and \(c_k\op d_{k+1}=c_{k-1}\).  In
particular \(p=c_{d-1}\), \(p\op a=q\) is the recurrence at \(k=d-1\), and
\(p\backslash q=a\).

Finally suppose \(d_j=x\).  By (T) with \((y,z)=(c_j,x)\),
\[
x=x\op((c_j\op x)\op c_j)=x\op(x\op c_j).
\]
Comparing with \(x=x\op(x\op q)\) and left-cancelling \(x\) twice gives \(c_j=q\).
Since \(q=c_{d-2}\) on the \(L_x\)-cycle, \(j\equiv d-2\pmod d\).
\end{proof}

\begin{corollary}[Small periods]\label{cor:small-periods}
If \(d=1\), then \(\EqAt{\E{255}}{x}\) holds.  The cases \(d=2\) and \(d=3\) are
impossible.  Thus a bad point has \(d\ge4\).
\end{corollary}
\begin{proof}
\cert \(d=1\) means \(x\op x=x\), so \(q=x\) and \(q\op x=x\).

If \(d=2\), then \(q=c_{d-2}=c_0=x\), hence \(d_1=q\op x=x\), contradicting the
forced-index statement in Proposition~\ref{prop:basic-orbit}, since \(1\not\equiv0
\pmod 2\).

Assume \(d=3\).  Then
\[
c_0=x,\qquad c_1=a=q,\qquad c_2=p,
\]
and
\[
a\op x=a,\qquad x\op a=p,\qquad x\op p=x,\qquad p\op a=a.
\]
The orbit recurrence at \(k=1\) gives
\[
x=a\op(p\op x). \tag{D3.1}
\]
By (K) with \((y,z)=(x,a)\),
\[
a=(x\op(x\op a))\op x
 =a\op(((x\op a)\op a)\op(x\op a))
 =a\op(a\op p).
\]
Since also \(a\op x=a\), left cancellation by \(a\) gives
\[
a\op p=x. \tag{D3.2}
\]
Comparing (D3.1) with (D3.2) and left-cancelling \(a\) gives
\[
p\op x=p. \tag{D3.3}
\]
Now (K) with \((y,z)=(p,a)\) gives
\[
x=(p\op(p\op a))\op p
 =a\op(((p\op a)\op a)\op(p\op a))
 =a\op((a\op a)\op a).
\]
Since \(a\op p=x\), left cancellation by \(a\) gives
\[
(a\op a)\op a=p. \tag{D3.4}
\]
Finally, (K) with \((y,z)=(a,x)\) gives
\[
(a\op a)\op a=x\op(a\op a).
\]
Using (D3.4) and \(x\op a=p\), left cancellation by \(x\) gives \(a\op a=a\).  Then
\((a\op a)\op a=a\), contradicting (D3.4) and the distinctness of \(a=c_1\) and \(p=c_2\)
on a period-three orbit.
\end{proof}

\begin{lemma}[Bad orbit-edge exclusions]\label{lem:bad-orbit-exclusions}
Assume \(\EqAt{\E{255}}{x}\) fails.  Then for every \(k\),
\[
d_k=c_k\op x\notin\{x,c_{k-1}\},
\]
or equivalently
\[
c_k\op x\neq x,
\qquad
x\op(c_k\op x)\neq c_k.
\]
\end{lemma}
\begin{proof}
\cert The first exclusion is Proposition~\ref{prop:unique-witness}: under failure, no
element \(u\) satisfies \(u\op x=x\).  If \(d_k=c_{k-1}\), then
\[
H_x(c_{k-1})=((x\op c_{k-1})\op x)=c_k\op x=d_k=c_{k-1},
\]
contradicting the fixed-point-free conclusion of Lemma~\ref{lem:fixed-map}.  Since
\(x\op c_{k-1}=c_k\), left cancellation by \(x\) makes \(d_k\neq c_{k-1}\) equivalent
to \(x\op d_k\neq c_k\).
\end{proof}

\begin{lemma}[Uniform orbit-edge identities]\label{lem:uniform-orbit}
For all \(k\),
\[
d_{k+2}=c_k\op\bigl((c_{k+1}\op c_k)\op c_{k+1}\bigr), \tag{U1}
\]
\[
x=d_k\op\bigl((c_k\op d_k)\op c_k\bigr), \tag{U2}
\]
\[
(c_k\op d_k)\op c_k=x\op\bigl((d_k\op x)\op d_k\bigr). \tag{U3}
\]
Therefore
\[
d_k\backslash x=(c_k\op d_k)\op c_k
=x\op\bigl((d_k\op x)\op d_k\bigr).
\]
At a bad point,
\[
a\backslash x=d_2,\qquad
q\backslash x=(a\op q)\op a=x\op((q\op x)\op q),\qquad
q\backslash a=(p\op q)\op p.
\]
\end{lemma}
\begin{proof}
\cert (U1) is (K) with \((y,z)=(x,c_k)\).  (U2) is (T) with \((y,z)=(c_k,x)\).
(U3) is (K) with \((y,z)=(c_k,x)\).  The quotient forms read (U2) as left division and
use (U3).

For the displayed specializations, \(a\backslash x=x\op((a\op x)\op a)=x\op(q\op a)\),
and (U1) at \(k=0\) says \(d_2=x\op(q\op a)\).  Since \(d_1=q\), the general quotient
formula at \(k=1\) gives
\[
q\backslash x=(a\op q)\op a=x\op((q\op x)\op q).
\]
Finally \(p\op a=q\), and (T) with \((y,z)=(p,a)\) gives
\[
a=q\op((p\op q)\op p),
\]
so \(q\backslash a=(p\op q)\op p\).
\end{proof}

\begin{proposition}[Period-four gate]\label{prop:period-four}
Assume \(d=4\).  Write
\[
a=c_1,\qquad q=c_2,\qquad p=c_3,\qquad r=q\op x,\qquad s=p\op x.
\]
Then
\[
x\op a=q,\quad a\op x=q,\quad a\op r=x,\quad x\op q=p,\quad x\op p=x,\quad p\op a=q,
\]
and \(r\notin\{x,a,q\}\).  Thus \(r=p\) or \(r\notin\{x,a,q,p\}\).  Further,
\[
q\op s=a,\qquad (p\op q)\op p=s.
\]
If \(r=p\), then
\[
q\op a=q,\qquad s=(q\op q)\op q=a\op(q\op q),\qquad s\notin\{x,a,q\},
\]
\[
H_x(a)=p,\qquad H_x(p)=a.
\]
If \(\EqAt{\E{255}}{x}\) fails and \(d=4\), then either
\[
q\op x\notin\{x,a,q,p\},
\]
or
\[
q\op x=p\qquad\text{and}\qquad p\op x\notin\{x,a,q,p\}.
\]
In particular, at least one of \(q\op x\) and \(p\op x\) lies outside
\(\{x,a,q,p\}\).
\end{proposition}
\begin{proof}
\cert The \(L_x\)-orbit identities give
\[
x\op a=q,\qquad x\op q=p,\qquad x\op p=x.
\]
Since \(q=d_1=a\op x\), we also have \(a\op x=q\).  The recurrence
\(c_{k-1}=c_k\op d_{k+1}\), for \(k=1,2,3\), gives
\[
a\op r=x,\qquad q\op s=a,\qquad p\op a=q.
\]

We next exclude \(r=x,a,q\).  If \(r=x\), then \(a\op r=a\op x=q\), but
\(a\op r=x\), so \(q=x\), impossible on a period-four orbit.

If \(r=a\), then (U3) at \(k=0\) gives
\[
a=r=x\op(q\op a).
\]
Since \(x\op x=a\), left cancellation by \(x\) gives \(q\op a=x\).  Now (K) with
\((y,z)=(q,x)\) gives
\[
(q\op r)\op q=x\op((r\op x)\op r).
\]
With \(r=a\), this becomes
\[
(q\op a)\op q=x\op((a\op x)\op a),
\]
hence \(p=a\), because the left side is \(x\op q=p\) and the right side is
\(x\op(q\op a)=x\op x=a\).  This is impossible.

If \(r=q\), then \(a\op q=x\).  By (U3) at \(k=1\),
\[
(a\op q)\op a=x\op((q\op x)\op q),
\]
so \(q=x\op(q\op q)\).  Comparing with \(q=x\op a\) gives \(q\op q=a\).  Applying (K)
with \((y,z)=(q,x)\) and using \(r=q\) gives
\[
(q\op q)\op q=x\op(q\op q).
\]
The left side is \(a\op q=x\), while the right side is \(x\op a=q\), another
contradiction.  Thus \(r\notin\{x,a,q\}\), so either \(r=p\) or \(r\) is outside the
principal orbit \(\{x,a,q,p\}\).

The identity \(q\op s=a\) was obtained above, and
\[
(p\op q)\op p=s
\]
is the specialization \(q\backslash a=(p\op q)\op p\) from
Lemma~\ref{lem:uniform-orbit}, because \(q\op s=a\).

Assume now that \(r=p\).  From (U3) at \(k=0\),
\[
p=r=x\op(q\op a).
\]
Since \(x\op q=p\), left cancellation by \(x\) gives
\[
q\op a=q. \tag{P4.1}
\]
Then (U1) at \(k=1\) gives
\[
s=a\op((q\op a)\op q)=a\op(q\op q). \tag{P4.2}
\]
Also (K) with \((y,z)=(q,a)\), together with (P4.1), gives
\[
(q\op q)\op q=a\op((q\op a)\op q)=a\op(q\op q)=s. \tag{P4.3}
\]
The exclusions for \(s\) are immediate.  If \(s=x\), then \(p\op x=x\), so
Proposition~\ref{prop:unique-witness} gives \(p=q\).  If \(s=a\), then
\[
a=q\op s=q\op a=q
\]
by (P4.1).  If \(s=q\), then \(p\op x=q=p\op a\), and left cancellation by \(p\) gives
\(x=a\).  Hence \(s\notin\{x,a,q\}\).

Finally,
\[
H_x(a)=((x\op a)\op x)=q\op x=r=p,\qquad
H_x(p)=((x\op p)\op x)=x\op x=a.
\]

If \(\EqAt{\E{255}}{x}\) fails, then \(r=q\op x\neq x\).  The preceding exclusions give
either \(r\notin\{x,a,q,p\}\), in which case we are done, or \(r=p\).  In the latter
case the exclusions above already give \(s=p\op x\notin\{x,a,q\}\).  It remains only
to exclude \(s=p\).  Suppose \(s=p\), and put \(t=q\op q\).  Then (P4.2) and (P4.3)
give
\[
a\op t=p,\qquad t\op q=p.
\]
Also \(p\op a=q\), and \(a\op p=x\) follows from \(a\op r=x\) together with \(r=p\).
Lemma~\ref{lem:division-trap}, applied with
\[
y=a,\qquad z=t,
\qquad u=p,
\qquad v=q,
\]
therefore gives
\[
t=a\op p=x.
\]
But then \(p=a\op t=a\op x=q\), contradicting the distinctness of \(p\) and \(q\) on
the period-four orbit.  Hence \(s\neq p\), and so \(s\notin\{x,a,q,p\}\).
\end{proof}

\begin{lemma}[Forward pair dynamics]\label{lem:phi-dynamics}
Define
\[
\Phi(u,v)=\bigl(u\op v,\ (u\op(u\op v))\op u\bigr).
\]
If \(\Phi(u,v)=(p,q)\), then
\[
p\op q=v,\qquad
\Phi(p,q)=\bigl(v,\ (p\op v)\op p\bigr).
\]
Consequently every forward \(\Phi\)-orbit can be written as
\[
\Phi^n(u,v)=(\alpha_n,\alpha_{n+2}),
\qquad
\alpha_n\op\alpha_{n+2}=\alpha_{n+1}
\]
for all \(n\ge0\), after setting
\[
\alpha_0=u,\qquad \alpha_2=v,\qquad
\alpha_1=u\op v,\qquad
\alpha_3=(u\op(u\op v))\op u.
\]
\end{lemma}
\begin{proof}
\cert If \(p=u\op v\) and \(q=(u\op(u\op v))\op u\), then (T) with
\((y,z)=(u,v)\) gives
\[
v=(u\op v)\op((u\op(u\op v))\op u)=p\op q.
\]
Therefore
\[
\Phi(p,q)=\bigl(p\op q,\ (p\op(p\op q))\op p\bigr)
          =\bigl(v,\ (p\op v)\op p\bigr).
\]
The displayed recurrence follows by applying the same calculation inductively to each
successive pair in the forward orbit.
\end{proof}

\begin{remark}[Expert collision route]\label{rem:expert-collision-route}
The expert input proposes the following finite collision lemma as the final fixed-point
gate:
\[
\boxed{
a\op x=b\op x=e,\quad a\neq b
\quad\Longrightarrow\quad
e\op x=x.}
\tag{C}
\]
If (C) is proved, then the finite implication follows immediately.  Indeed, if
\(\EqAt{\E{255}}{x}\) fails, Proposition~\ref{prop:unique-witness} gives no solution to
\(u\op x=x\).  Thus \(x\notin\operatorname{im}R_x\), so finiteness makes \(R_x\)
non-injective.  Choose \(a\neq b\) with \(a\op x=b\op x=e\).  Then (C) supplies
\(e\op x=x\), contradicting the missing right fixed point over \(x\).

The algebraic reduction behind (C) is already aligned with the identities above.  From
(T), first with \((y,z)=(a,x)\) and then with \((y,z)=(b,x)\),
\[
x=e\op((a\op e)\op a)=e\op((b\op e)\op b).
\]
Left cancellation by \(e\) gives a common element
\[
t=(a\op e)\op a=(b\op e)\op b,\qquad e\op t=x,
\]
and hence
\[
\Phi(a,x)=\Phi(b,x)=(e,t).
\]
Thus (C) is equivalent, for this collision, to ruling out \(t\neq x\).
Lemma~\ref{lem:phi-dynamics} supplies the proposed finite engine: if \(t\neq x\), the
forward \(\Phi\)-orbit of \((e,t)\) eventually repeats, and the intended last step is to
pull the first repeated pair backwards along
\[
\alpha_n\op\alpha_{n+2}=\alpha_{n+1}
\]
until it reaches the two predecessors \((a,x)\) and \((b,x)\).  This is now the exact
audit point: the pullback must be justified using only proved left cancellation, never
right cancellation by the second coordinate.  Once that finite pullback is formalized,
(C) closes the proof of \(\E{677}\models_{\fin}\E{255}\).
\end{remark}

\begin{proposition}[Collision lift]\label{prop:collision-lift}
If \(d_i=d_j=e\), then
\[
(c_i\op e)\op c_i=(c_j\op e)\op c_j.
\]
If also \(i\not\equiv j\pmod d\), then
\[
c_i\op e\neq c_j\op e.
\]
\end{proposition}
\begin{proof}
\cert (T) gives \(x=d_k\op((c_k\op d_k)\op c_k)\).  Equal \(d_i,d_j\) allow left
cancellation by \(e\).  If also \(c_i\op e=c_j\op e=s\), then left-cancel \(s\) to get
\(c_i=c_j\).
\end{proof}

\begin{proposition}[Right-collision rectangles]\label{prop:rectangles}
If \(y\op x=r\), then
\[
(y\op r)\op y=r\backslash x. \tag{R}
\]
Consequently, if \(y\op x=z\op x=r\) and \(y\op r=z\op r\), then \(y=z\).
\end{proposition}
\begin{proof}
\cert (T) with \((y,z)=(y,x)\) gives \(x=r\op((y\op r)\op y)\).  This is (R).  The
last claim is left cancellation after applying (R) to \(y\) and \(z\).
\end{proof}

\begin{lemma}[Right-tail first repetition]\label{lem:right-tail}
Assume \(q\op x\neq x\).  Define
\[
\rho_0=x,\qquad \rho_{n+1}=\rho_n\op x.
\]
Then \(\rho_n\neq x\) for \(n\ge1\).  Let \(1\le i<j\) be the first repetition in the
tail, with \(j\) minimal, and put
\[
e=\rho_i=\rho_j,\qquad y=\rho_{i-1},\qquad z=\rho_{j-1}.
\]
Then
\[
y\neq z,\qquad y\op x=z\op x=e,
\]
\[
(y\op e)\op y=(z\op e)\op z=e\backslash x,\qquad y\op e\neq z\op e.
\]
\end{lemma}
\begin{proof}
\cert If some \(\rho_n=x\), then \(\rho_{n-1}\op x=x\), forcing
\(\rho_{n-1}=q\), contradiction.  Finiteness gives a first repetition.  Minimality gives
\(y\neq z\).  Proposition~\ref{prop:rectangles} gives the rectangle and splitter.
\end{proof}

\begin{proposition}[Left-translation labels]\label{prop:labels}
Write \(a\op b=\sigma(a)(b)\).  Then \(\sigma(a)\in\operatorname{Sym}(M)\) and the
label map \(a\mapsto\sigma(a)\) is injective.  If \(y\op x=z\op x=r\) with \(y\neq z\),
then
\[
\sigma(y)(x)=\sigma(z)(x)=r,\qquad \sigma(y)(r)\neq\sigma(z)(r),
\]
\[
\sigma(\sigma(y)(r))(y)=r\backslash x=
\sigma(\sigma(z)(r))(z).
\]
\end{proposition}
\begin{proof}
\cert Left translations are permutations by Proposition~\ref{prop:left-division}.  Suppose
\(L_y=L_z\).  Choose any \(x\in M\) and put
\[
e=y\op x=z\op x.
\]
The two \(\E{677}\) specializations give
\[
x=y\op(x\op(e\op y)),\qquad x=z\op(x\op(e\op z)).
\]
Since \(L_y=L_z\), the second equality may be written as
\[
x=y\op(x\op(e\op z)).
\]
Comparing with the first equality and left-cancelling successively by \(y\), then by
\(x\), then by \(e\), gives \(y=z\).  Thus \(a\mapsto\sigma(a)\) is injective.

If \(y\op x=z\op x=r\) and \(y\neq z\), Proposition~\ref{prop:rectangles} gives
\[
(y\op r)\op y=(z\op r)\op z=r\backslash x.
\]
It also says that \(y\op r=z\op r\) would force \(y=z\), so
\(\sigma(y)(r)\neq\sigma(z)(r)\).  Rewriting these products in the notation
\(a\op b=\sigma(a)(b)\) gives the displayed formulas.
\end{proof}

\begin{proposition}[Universal stress tests]\label{prop:universal-stress-tests}
Assume \(\EqAt{\E{255}}{x}\) fails.  Then every candidate around \(x\) must pass the
following tests.
\begin{enumerate}
\item \(R_x\) is not surjective and not injective.  In particular, no finite
right-cancellative, quasigroup, or loop candidate can be bad at \(x\).
\item No commutative finite candidate can be bad at \(x\).
\item The \(L_x\)-period of \(x\) is \(d\ge4\).  In particular, \(x\op x\neq x\), and
the periods \(d=2\) and \(d=3\) are unavailable.
\item For every orbit edge,
\[
d_k=c_k\op x\notin\{x,c_{k-1}\}.
\]
\item Every first repetition in the right tail \(\rho_{n+1}=\rho_n\op x\) produces a
splitter rectangle: if \(y\op x=z\op x=e\) with \(y\neq z\), then
\[
(y\op e)\op y=(z\op e)\op z=e\backslash x,
\qquad
y\op e\neq z\op e.
\]
\item Every proposed short return must pass the division return trap of
Lemma~\ref{lem:division-trap}: if \(u=y\op z\), \(v=u\op y\), and \(z\op v=u\), then
\(z=y\op u\).
\end{enumerate}
\end{proposition}
\begin{proof}
\cert{} At a bad point, Proposition~\ref{prop:unique-witness} gives no solution to
\(u\op x=x\), so \(x\notin\operatorname{im}R_x\).  Since the magma is finite, an
injective \(R_x\) would be surjective.  This proves the first item.  If the operation is
commutative, then \(R_x=L_x\), but \(L_x\) is bijective by
Proposition~\ref{prop:left-division}; this proves the second item.  The period statement
is Corollary~\ref{cor:small-periods}.  The orbit-edge exclusion is
Lemma~\ref{lem:bad-orbit-exclusions}.  The splitter rectangle is
Lemma~\ref{lem:right-tail}, or Proposition~\ref{prop:rectangles} for any chosen
collision fiber.  The last item is Lemma~\ref{lem:division-trap}.
\end{proof}

\begin{proposition}[Period-five internal-tail stress tests]\label{prop:p5-internal-tests}
Assume \(d=5\) and \(\EqAt{\E{255}}{x}\) fails.  Write
\[
a=c_1,\qquad b=c_2,\qquad q=c_3,\qquad p=c_4,
\]
and put
\[
e=b\op x,
\qquad g=q\op x,
\qquad s=p\op x.
\]
Then
\[
e\notin\{x,a\},\qquad g\notin\{x,b\},\qquad s\notin\{x,q\},
\]
and
\[
b\op g=a,
\qquad q\op s=b,
\qquad p\op a=q. \tag{P5.0}
\]
If the right tail stays in the principal orbit through \(g\), then the only possibilities
are \(g\in\{a,q,p\}\), and they have the following stress data.
\begin{enumerate}
\item If \(g=q\), then
\[
b\op q=a,
\qquad
(a\op q)\op a=(q\op q)\op q=q\backslash x,
\qquad
a\op q\neq q\op q.
\]
\item If \(g=a\), then
\[
b\op a=a,
\qquad
(q\op a)\op q=b\op x=a\backslash x,
\qquad
q\op a\neq b.
\]
\item If \(g=p\), then
\[
b\op p=a,
\qquad
q\op s=b,
\qquad
s\in M\setminus\{x,q\}.
\]
In the internal subcases for \(s\), the following extra tests apply:
\[
\begin{array}{ccl}
s=a&\Longrightarrow&q\op a=b,\\[-1mm]
&&q\op p=b\op x=a\backslash x,\\[1mm]
s=p&\Longrightarrow&q\op p=b,\\[-1mm]
&&b\op q=(p\op p)\op p=p\backslash x,
\quad q\op p\neq p\op p,\\[1mm]
s=b&\Longrightarrow&q\op b=b.
\end{array}
\]
If moreover \(s=b\) and the next value \(e=b\op x\) is still principal-orbit-internal,
then \(e\in\{b,q,p\}\), giving respectively the collision fibers
\[
\{p,b\}\to b,
\qquad
\{a,b\}\to q,
\qquad
\{q,b\}\to p.
\]
\end{enumerate}
\end{proposition}
\begin{proof}
\cert{} The exclusions are Proposition~\ref{prop:p5p6}.  The recurrence
\(c_{k-1}=c_k\op d_{k+1}\), with \(d_3=g\) and \(d_4=s\), gives
\[
b\op g=a,
\qquad q\op s=b,
\]
and the recurrence at \(k=4\) gives \(p\op a=q\).

If \(g=q\), then \(b\op q=a\) follows from \(b\op g=a\).  Also
\(a\op x=q=q\op x\), so Proposition~\ref{prop:rectangles}, applied to the fiber
\(\{a,q\}\to q\), gives the displayed splitter rectangle.

If \(g=a\), then \(b\op a=a\).  Since \(x\op x=a=q\op x\),
Proposition~\ref{prop:rectangles} gives
\[
(x\op a)\op x=(q\op a)\op q=a\backslash x,
\qquad x\op a\neq q\op a.
\]
Because \(x\op a=b\), this is the displayed statement.

If \(g=p\), then \(b\op p=a\), and the relation \(q\op s=b\) is already in (P5.0).
The exclusion \(s\notin\{x,q\}\) is the bad orbit-edge exclusion at \(k=4\).  If
\(s=a\), then \(q\op a=b\), and the fiber \(\{x,p\}\to a\) gives
\[
(x\op a)\op x=(p\op a)\op p=a\backslash x.
\]
Using \(x\op a=b\) and \(p\op a=q\), this becomes \(q\op p=b\op x=a\backslash x\).
If \(s=p\), then \(q\op p=b\), and the fiber \(\{q,p\}\to p\) gives
\[
(q\op p)\op q=(p\op p)\op p=p\backslash x,
\qquad q\op p\neq p\op p,
\]
which is the displayed statement.  If \(s=b\), then \(q\op b=b\).  Finally, when
\(s=b\), the next right-tail value is \(e=b\op x\).  If it is principal-orbit-internal,
then \(e\notin\{x,a\}\), so \(e\in\{b,q,p\}\); comparing with the earlier right-tail
values \(a\op x=q\), \(q\op x=p\), and \(p\op x=b\) gives the three listed fibers.
\end{proof}

\begin{remark}[Candidate pruning ledger]\label{rem:candidate-ledger}

the cited stress tests; a surviving line is only a local blueprint for the next pruning
pass.
\begin{enumerate}[label=\textup{C\arabic*.},leftmargin=2.7em]

\begin{array}{ccl}
\EqAt{\E{255}}{x}
&\Longleftrightarrow& q\op x=x \\[1mm]
&\Longleftrightarrow& \exists u\ (u\op x=x) \\[1mm]
&\Longleftrightarrow& q\backslash x=x \\[1mm]
&\Longleftrightarrow& ((x\op x)\op q)\op(x\op x)=x \\[1mm]
&\Longleftrightarrow& (q\op x)\op q=p .
\end{array}
\]
Under the negation \(q\op x\neq x\):
\[
\begin{array}{ccl}
\text{no }u\op x=x
&\Longrightarrow& x\notin \operatorname{im}R_x \\[1mm]
&\Longrightarrow& R_x\text{ has a nontrivial fiber} \\[1mm]
&\Longrightarrow& y\op x=z\op x=e,\ y\neq z \\[1mm]
&\Longrightarrow& (y\op e)\op y=(z\op e)\op z=e\backslash x,\quad
y\op e\neq z\op e .
\end{array}
\]
The canonical way to choose the fiber is the first repetition in the right tail
\(\rho_{n+1}=\rho_n\op x\).  This removes choice from the splitter route.

For the orbit route:
\[
\begin{array}{ccl}
q\op x\neq x
&\Longrightarrow& d\ge4,\\[1mm]
d=4
&\Longrightarrow& q\op x\in\{p\}\cup(M\setminus\{x,a,q,p\}),\\[1mm]
d=4,\ q\op x=p
&\Longrightarrow& p\op x\notin\{x,a,q,p\}.
\end{array}
\]
So a bad period-four point exits the principal orbit along the right tail by the step
\(p\op x\).  No four-element principal-orbit-only bad point survives these facts.
\end{remark}

\begin{remark}[Finite facts used, nothing else]
Only these finite principles appear: finite surjection \(=\) bijection; a finite
self-map missing a point has a nontrivial fiber; every finite forward orbit repeats; and
conjugate finite self-maps have the same cycle structure.  Every cancellation in the
paper is left cancellation by a proved bijective left translation.
\end{remark}

\begin{remark}[Do not use]
\[
\boxed{
\begin{gathered}
\text{No right cancellation. No associativity. No identity element.}\\
\text{No universal idempotence. No global ``\(H_x\) has no cycles''.}\\
\text{Do not use } d_k\backslash x=c_{k-2}\op c_{k-1}.\\
\text{Do not use } p\op d_k=d_{k+1}\text{ unless freshly proved.}\\
\text{Do not pull back a \(\Phi\)-repeat by cancelling a right coordinate.}
\end{gathered}}
\]
\end{remark}

\begin{target}[Expert collision gate]\label{target:expert-collision}
Complete the finite pullback step in Remark~\ref{rem:expert-collision-route}: prove that
if
\[
a\op x=b\op x=e,\qquad a\neq b,
\]
then \(e\op x=x\).  Equivalently, with
\[
t=(a\op e)\op a=(b\op e)\op b,\qquad e\op t=x,
\]
rule out \(t\neq x\) by the finite \(\Phi\)-orbit recurrence from
Lemma~\ref{lem:phi-dynamics}.
\end{target}

\begin{target}[Fixed-point gate]
Prove \(q\op x=x\), equivalently Lemma L or \((q\op x)\op q=p\), with a visible finite
step.
\end{target}

\begin{target}[Splitter gate]
Use the canonical right-tail collision fiber, or any collision \(d_i=d_j=e\), to turn
the injective splitter \(t\mapsto t\op e\) into either \(q\op x=x\) or a finite counting
contradiction.
\end{target}

\begin{target}[Period-four external gate]
At a bad period-four point, feed the forced external element among
\[
q\op x,\qquad p\op x
\]
back into the fixed-point gate or the splitter gate.
\end{target}

\begin{target}[Period-five pruning gate]
Use the explicit internal splitter rectangles in Proposition~\ref{prop:p5-internal-tests},
especially the branches \(q\op x=q\) and \(q\op x=a\), to force either \(q\op x=x\), an
external product, or a finite counting contradiction.
\end{target}
